% this document is the framework setting the general layout of the card, importing the required packages to customise it.

% the first line is about declaring a bigger (20pt) font size, we do this by using the 'extarticle' class - supporting sizes 8pt, 9pt, 10pt, 11pt, 12pt, and 20pt. You can find more here: https://www.ctan.org/tex-archive/macros/latex/contrib/extsizes?lang=en.
\documentclass[12pt]{extarticle}

% utf8 enconding support.
\usepackage[utf8]{inputenc}

% the geometry packacge let's you use a custom paper size.
\usepackage[a5paper]{geometry}
% in this example portrait orientation (width 59mm, height 88mm), this should be the full bleed sizes.
\geometry{paperwidth=59mm,paperheight=88mm}
% here you set margin, the distance from paper edges to where you'll have your content
\geometry{left=8mm,top=40pt,right=4mm,bottom=4mm}

% the xcolor package to support custom colours
\usepackage{xcolor}
% colours definition, these are nothing else than aliases, so you can use just the colour name. CMYK colour space is used her, but you can use RGB and few others. You can find more here: https://www.ctan.org/pkg/xcolor?lang=en
\definecolor[named]{grey}{cmyk}{0,0,0,0.27}
\definecolor[named]{dgrey}{cmyk}{0,0,0,0.72}
\definecolor[named]{orange}{cmyk}{0,0.8,1,0}

% the fontspec package support for all fonts you have on your system. Please note, you'll have to use XeTeX to print your PDF to use your own font. Find more here: http://tug.org/xetex/
\usepackage{fontspec}
% here I'm setting the default font for the document.
\setmainfont{Helvetica Neue LT Std 97 Black Condensed}

% the setspace package support lineheight customisation for paragraphs.
\usepackage{setspace}

% setting header, and footer to blank.
\pagestyle{empty}

% starting the document, 'importing' the actual content to keep it separeted as much as possible.
\begin{document}
\include{bc_back}
\end{document}
